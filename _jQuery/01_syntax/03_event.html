Element.onEvent = function(){...} => $('selector').eventName(function(){...})

Element.addEventListener('eventName', function(){...}) => $('selector').on('eventName', function({...}))

  • on() 메서드를 많이 사용하고 있고 다양한 장점이 있다
  • on() 메서드는 파라미터로 이벤트 발생주체를 지정할 수 있다
  • 또한 파라미터로 실행할 함수에 데이터를 바인딩시켜 전달 할 수 있다
  • 여러 이벤트를 객체타입으로 선언하여 파라미터로 던질 수 있다
